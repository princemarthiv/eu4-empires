# The South's Response
country_event = {
	id = german_unification.3
	title = "german_unification.EVTNAME3"
	desc = {
		trigger = {
			OR = {
				tag = HAB
				tag = HEM
				tag = AHE
			}
		}
		desc = "german_unification.EVTDESC3.a"
	}
	desc = {
		trigger = {
			NOT = {
				tag = HAB
				tag = HEM
				tag = AHE
			}
		}
		desc = "german_unification.EVTDESC3.b"
	}	
	picture = CONQUEST_eventPicture
	fire_only_once = yes
	major = yes
	trigger = {
		NGF = {
			num_of_owned_provinces_with = {
				region = north_german_region
				value = 40
			}
		}
		OR = {
			tag = HAB
			tag = HEM
			tag = AHE
			any_country = {
				capital_scope = {
					region = south_german_region 
				}
				num_of_owned_provinces_with = {
					region = south_german_region
					value = 20
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 1
	}
		
	option = {
		name = "german_unification.EVT3OPTA"
		if = {
			limit = {
				OR = {
					tag = HAB
					tag = HEM
					tag = AHE
				}
			}
			hidden_effect = {
				country_event = { 
					id = german_unification.5 
					days = 30
				}
			}
		}
		if = {
			limit = {
				NOT = {
					tag = HAB
					tag = HEM
					tag = AHE
				}
			}
			change_tag = SGF
		}
		every_country = {
			limit = {
				capital_scope = { region = south_german_region }
				NOT = {
					tag = HAB
					tag = AHE
					tag = HEM
				}
			}
			country_event = { id = german_unification.4 }
		}
		ai_chance = {factor = 1}
	}
	
	option = {
		name = german_unification.EVT3OPTB
		add_prestige = -10
		ai_chance = {factor = 0}
	}
}
